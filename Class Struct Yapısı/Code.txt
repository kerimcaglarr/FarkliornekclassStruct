struct Kare{

    

    var kenarUzunlugu:Int

    

    func cevrehesapla()->Int{

        

        return kenarUzunlugu*4

    }


    func alanHesapla()->Int{

        return kenarUzunlugu*kenarUzunlugu

    }

}

kenariUcOlanKare.alanHesapla()
